2. 태그 
	<% 자바 코드 %>
	<%= %>
	<%-- --%> : jsp 주석 : 클라이언트로 전송되지 않는다. 
	<%@ page
		<%@ page import="" %>
	
	<%@ include : 페이지를 이동시키는 태그

	<jsp:include : 페이지를 이동시키는 액션태그 
	<jsp:param : 페이지 이동시 데이터를 전송하는 액션태그
